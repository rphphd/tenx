'use strict';
 
/**
 * @ngdoc directive
 * @name re2App.directive:plistHeader
 * @description
 * # plistHeader
 */
angular.module('re2App')
  .directive('plistHeader', function ($compile) {
    var theTemplate = '';
 
    var returnObject = {
      restrict: 'E',
      templateUrl: function(elem,attrs) {
        return attrs.headerView;
      },
      link: function postLink(scope, element, attrs) {
        //element.html('this is the plistHeader directive');
        scope.$watch('attrs.headerView', function(newValue){
          attrs.headerView=scope.headerView;
          console.log('plistHeader',scope, element, attrs, theTemplate);
          $compile(element.contents())(scope);
        });
      }
    };
 
    return returnObject;
  });
